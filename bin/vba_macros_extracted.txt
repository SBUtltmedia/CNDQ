Private Sub CommandButton1_Click()

Dim X As Single
Dim Done  As Boolean
Dim Msg As String
Dim J As Integer
Dim S As Integer

Worksheets(ComboBox1.Value).Select
X = TextBox1.Value
Done = False

Select Case ComboBox3.Value
    Case "C"
        If Range("F5").Value < X Then
            Msg = ComboBox1.Value & " does not have " & X & " gallons of Liquid C available."
            MsgBox (Msg)
            Done = True
        End If
    Case "N"
        If Range("F6").Value < X Then
            Msg = ComboBox1.Value & " does not have " & X & " gallons of Liquid N available."
            MsgBox (Msg)
            Done = True
        End If
    Case "D"
        If Range("F7").Value < X Then
            Msg = ComboBox1.Value & " does not have " & X & " gallons of Liquid D available."
            MsgBox (Msg)
            Done = True
        End If
    Case "Q"
        If Range("F8").Value < X Then
            Msg = ComboBox1.Value & " does not have " & X & " gallons of Liquid Q available."
            MsgBox (Msg)
            Done = True
        End If
End Select

If Done = False Then
    J = 1
    Do
        If Range("N5").Offset(J, 0).Value = "" Then
            Range("N5").Offset(J, 0).Value = ComboBox2.Value
            Select Case ComboBox3.Value
                Case "C"
                    Range("J5").Offset(J, 0).Value = TextBox1.Value
                    Range("O5").Offset(J, 0).Value = TextBox2.Value
                Case "N"
                    Range("K5").Offset(J, 0).Value = TextBox1.Value
                    Range("P5").Offset(J, 0).Value = TextBox2.Value
                Case "D"
                    Range("L5").Offset(J, 0).Value = TextBox1.Value
                    Range("Q5").Offset(J, 0).Value = TextBox2.Value
                Case "Q"
                    Range("M5").Offset(J, 0).Value = TextBox1.Value
                    Range("R5").Offset(J, 0).Value = TextBox2.Value
            End Select
            Done = True
        Else
            J = J + 1
        End If
    Loop Until Done
    
    S = SolverSolve(False)
    If S <> 0 Then
        MsgBox (S)
    End If
    
    Worksheets(ComboBox2.Value).Select
    Done = False
    J = 1
    Do
        If Range("Z5").Offset(J, 0).Value = "" Then
            Range("Z5").Offset(J, 0).Value = ComboBox1.Value
            Select Case ComboBox3.Value
                Case "C"
                    Range("V5").Offset(J, 0).Value = TextBox1.Value
                    Range("AA5").Offset(J, 0).Value = TextBox2.Value
                Case "N"
                    Range("W5").Offset(J, 0).Value = TextBox1.Value
                    Range("AB5").Offset(J, 0).Value = TextBox2.Value
                Case "D"
                    Range("X5").Offset(J, 0).Value = TextBox1.Value
                    Range("AC5").Offset(J, 0).Value = TextBox2.Value
                Case "Q"
                    Range("Y5").Offset(J, 0).Value = TextBox1.Value
                    Range("AD5").Offset(J, 0).Value = TextBox2.Value
            End Select
            Done = True
        Else
            J = J + 1
        End If
    Loop Until Done
    
    S = SolverSolve(False)
    If S <> 0 Then
        MsgBox (S)
    End If

End If

Worksheets("Scoreboard").Select

Call CommandButton2_Click

End Sub
